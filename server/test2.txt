    function encodeB64(str) { return btoa(unescape(encodeURIComponent(str))); }
    function decodeB64(str) { return decodeURIComponent(escape(atob(str))); }

    async function fetchGH(path) {
        try {
            const res = await fetch(`https://api.github.com/repos/${GH_USER}/${GH_REPO}/contents/${path}`, { headers: GH_TOKEN ? { "Authorization": `token ${GH_TOKEN}` } : {} });
            if (res.ok) { const d = await res.json(); return { content: decodeB64(d.content), sha: d.sha }; }
            return null;
        } catch(e) { return null; }
    }

    async function saveGH(path, content, msg, sha) {
        if (!GH_TOKEN) { alert("GitHub Token missing! Configure in Settings."); return null; }
        const body = { message: msg, content: encodeB64(content), branch: "main" };
        if (sha) body.sha = sha;
        try {
            const res = await fetch(`https://api.github.com/repos/${GH_USER}/${GH_REPO}/contents/${path}`, {
                method: "PUT", headers: { "Authorization": `token ${GH_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify(body)
            });
            if (res.ok) { const d = await res.json(); return d.content.sha; }
            else { const err = await res.json(); alert("Save Error: " + err.message); return null; }
        } catch(e) { alert("Network Error"); return null; }
    }

    async function uploadImageGH(file) {
        if (!GH_TOKEN) { alert("GitHub Token missing!"); return null; }
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.onload = async (e) => {
                const b64 = e.target.result.split(',')[1];
                const ext = file.name.split('.').pop() || 'png';
                const path = `storage/images/${Date.now()}.${ext}`;
                const body = { message: `Upload image ${file.name}`, content: b64, branch: "main" };
                try {
                    const res = await fetch(`https://api.github.com/repos/${GH_USER}/${GH_REPO}/contents/${path}`, {
                        method: "PUT", headers: { "Authorization": `token ${GH_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify(body)
                    });
                    if (res.ok) resolve(`https://raw.githubusercontent.com/${GH_USER}/${GH_REPO}/main/${path}`);
                    else reject("Upload failed");
                } catch (err) { reject(err); }
            };
            reader.readAsDataURL(file);
        });
    }

    async function runCounter() {
        var vcEl = document.getElementById('view-count');
        var statTotal = document.getElementById('stat-total');
        var statFreq = document.getElementById('stat-freq');
        
        try {
            var isVisited = localStorage.getItem(VISITED_KEY);
            var isReturned = localStorage.getItem('returned_webt');
            // Use VC_TOKEN per user request (fallback to GH_TOKEN if VC_TOKEN is placeholder and GH_TOKEN exists)
            var tokenToUse = (VC_TOKEN === 'YOUR_GITHUB_TOKEN' && GH_TOKEN) ? GH_TOKEN : VC_TOKEN;

            var res = await fetch(VC_API, {
                headers: { 'Accept': 'application/vnd.github+json', 'Authorization': 'token ' + tokenToUse }
            });
            if (!res.ok) throw new Error('HTTP ' + res.status);

            var data = await res.json();
            // Assign to appState to ensure Host edits don't overwrite stats
            appState.home = JSON.parse(decodeB64(data.content));
            appState.shas.home = data.sha;
            
            var currentVisitors = appState.home.visitors || 0;
            var currentFrequent = appState.home.frequent || 0;
            var needsWrite = false;

            if (!isVisited) {
                appState.home.visitors = currentVisitors + 1;
                localStorage.setItem(VISITED_KEY, '1');
                needsWrite = true;
            } else if (!isReturned) {
                appState.home.frequent = currentFrequent + 1;
                localStorage.setItem('returned_webt', '1');
                needsWrite = true;
            }

            if (needsWrite) {
                var writeRes = await fetch(VC_API, {
                    method: 'PUT',
                    headers: { 'Accept': 'application/vnd.github+json', 'Authorization': 'token ' + tokenToUse, 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        message: 'Update visitor stats',
                        content: encodeB64(JSON.stringify(appState.home, null, 2)),
                        sha: appState.shas.home
                    })
                });
                if (writeRes.ok) {
                    var newDat = await writeRes.json();
                    appState.shas.home = newDat.content.sha;
                }
            }

            if(vcEl) vcEl.textContent = (appState.home.visitors || 0).toLocaleString();
            if(statTotal) statTotal.textContent = (appState.home.visitors || 0).toLocaleString();
            if(statFreq) statFreq.textContent = (appState.home.frequent || 0).toLocaleString();

        } catch (err) {
            if (!appState.home.motto) appState.home = { motto: defaultMottoRaw, desc: defaultDescRaw, visitors: 0, frequent: 0 };
            if(vcEl) vcEl.textContent = '—';
            if(statTotal) statTotal.textContent = '—';
            if(statFreq) statFreq.textContent = '—';
        }
    }

    window.zenDefaultTime = 25 * 60;

    function parseZenInputs() {
        let h = parseInt(document.getElementById('zen-hr').value.replace(/\D/g, ''), 10) || 0;
        let m = parseInt(document.getElementById('zen-min').value.replace(/\D/g, ''), 10) || 0;
        let s = parseInt(document.getElementById('zen-sec').value.replace(/\D/g, ''), 10) || 0;
        
        if (h < 0) h = 0;
        if (m < 0) m = 0; 
        if (s < 0) s = 0;
        
        if (s > 59) { m += Math.floor(s/60); s = s % 60; }
        if (m > 59) { h += Math.floor(m/60); m = m % 60; }
        
        if (h === 0 && m === 0 && s === 0) m = 25; 
        if (h > 99) h = 99; 
        
        zenTimeLeft = (h * 3600) + (m * 60) + s;
        window.zenDefaultTime = zenTimeLeft; 
        updateZenTimerDisplay();
    }

    function updateZenTimerDisplay() {
        const h = Math.floor(zenTimeLeft / 3600).toString().padStart(2, '0');
        const m = Math.floor((zenTimeLeft % 3600) / 60).toString().padStart(2, '0');
        const s = (zenTimeLeft % 60).toString().padStart(2, '0');

        const minInput = document.getElementById('zen-min');
        const secInput = document.getElementById('zen-sec');

        if (minInput) minInput.value = m;
        if (secInput) secInput.value = s;
    }

    var audioCtx = null;
    function clickSound(f) { 
        try { 
            if (!window.AudioContext && !window.webkitAudioContext) return; 
            if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)(); 
            if (audioCtx.state === 'suspended') audioCtx.resume(); 
            var o = audioCtx.createOscillator(), g = audioCtx.createGain(); 
            o.frequency.value = f || 400; 
            g.gain.exponentialRampToValueAtTime(0.01, audioCtx.currentTime + 0.1); 
            o.connect(g); 
            g.connect(audioCtx.destination); 
            o.start(); 
            o.stop(audioCtx.currentTime + 0.1); 
        } catch (e) {} 
    }

    function fetchTimeout(url, ms) { 
        ms = ms || 8000; 
        var ctrl = new AbortController(), id = setTimeout(() => ctrl.abort(), ms); 
        var headers = GH_TOKEN ? { "Authorization": `token ${GH_TOKEN}` } : {};
        return fetch(url, {signal: ctrl.signal, headers: headers}).finally(() => clearTimeout(id));

    function initLibrary() {
        fetchTimeout('https://api.github.com/repos/' + SONG_USER + '/' + SONG_REPO + '/contents/' + SONG_BASE)
            .then(r => r.ok ? r.json() : []).then(data => {
                if (!Array.isArray(data)) throw new Error();
                var dirs = data.filter(f => f.type === 'dir'); 
                PLAYLISTS = dirs.length ? dirs.map(d => d.name) : [SONG_BASE]; 
                renderTabs();
                
                return Promise.all(PLAYLISTS.map(pl => fetchTimeout('https://api.github.com/repos/' + SONG_USER + '/' + SONG_REPO + '/contents/' + (dirs.length ? (SONG_BASE + '/' + pl) : SONG_BASE))
                    .then(r => r.ok ? r.json() : [])
                    .then(files => {
                        if (!Array.isArray(files)) return []; 
                        var audios = files.filter(f => /\.(mp3|wav|ogg|m4a)$/i.test(f.name));
                        var lrcs = files.filter(f => f.name.endsWith('.lrc'));
                        return audios.map(f => { 
                            var base = f.name.replace(/\.[^/.]+$/, '');
                            var lrc = lrcs.find(l => l.name.startsWith(base)); 
                            return { t: base.replace(/[-_]/g, ' '), file: f.name, playlist: pl, lrc: lrc ? lrc.name : null, key: pl + '/' + f.name }; 
                        });
                    }).catch(() => [])
                ));
            }).then(results => { 
                songs = [].concat.apply([], results); 
                renderLibrary(); 
                statusLabel.textContent = 'Ready'; 
                artistName.textContent = songs.length + ' tracks'; 
            })
            .catch(() => { 
                statusLabel.textContent = 'Sync Failed'; 
                artistName.textContent = 'Check repo / connection'; 
                libCount.textContent = 'Sync Failed'; 
            });
    }
